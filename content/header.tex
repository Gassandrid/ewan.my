% 1. Load the 'float' package to get the [H] placement option
\usepackage{float}

% 2. Redefine the 'figure' environment to always use [H] (Here)
% This stops LaTeX from moving your images around (floating)
\let\origfigure\figure
\let\endorigfigure\endfigure
\renewenvironment{figure}[1][]{%
  \origfigure[H]
}{%
  \endorigfigure
}

% 3. Set a global default width for all images
% This scales every image to 70% of the page's text width.
% You can change 0.7 to another value like 0.5 (50%) or 0.8 (80%).
\setkeys{Gin}{width=0.7\textwidth}
